return {
    {
        "folke/tokyonight.nvim",
        lazy = false,
        priority = 1000,
        opts = {
            -- The main theme style: "storm", "moon", "night", "day"
            style = "storm",
            -- Enable this to use the TokyoNight colors for Terminal (like alacritty/kitty)
            terminal_colors = true,
            -- Darken the background of "non-focused" windows
            dim_inactive = true,
            -- Set the style of the background:
            -- "dark" (default), "transparent" (if transparent = true), or "normal"
            transparent = false,
            -- Adds a different color to the floating windows and borders
            styles = {
                comments = { italic = true },
                keywords = { italic = true },
                functions = {},
                variables = {},
                -- Background styles. Can be "dark", "transparent" or "normal"
                sidebars = "dark", -- style for sidebars, like nvim-tree / neo-tree
                floats = "dark", -- style for floating windows
            },
            -- Sidebars that should be colored differently
            sidebars = { "qf", "help", "terminal", "packer" },
            -- Highlighting for the day flavor
            day_brightness = 0.3,
            -- When true, section headers in the lualine theme will be bold
            lualine_bold = false,

            --- @usage: Overriding colors/groups
            -- You can override specific color values
            on_colors = function(colors)
                -- Example: colors.hint = "#25D366"
                colors.bg = "#1a1b26" -- The "Night" deep blue-black
                colors.bg_dark = "#16161e" -- Even darker for sidebars
                colors.bg_float = "#16161e" -- Darker for floating windows
            end,
            -- You can override specific highlights
            on_highlights = function(highlights, colors)
                -- Example: highlights.TelescopeNormal = { bg = colors.bg_dark, fg = colors.fg_dark }
            end,

            -- Cache the theme to make it load even faster
            cache = true,
        },
    },

    -- This tells LazyVim to actually use the theme we just configured
    {
        "LazyVim/LazyVim",
        opts = {
            colorscheme = "tokyonight-storm",
        },
    },
}
